## 1. Purpose

Provide GitLab CI/CD integration templates and support for code-to-gate analysis in GitLab pipelines.

---

## 2. Scope

### Included
- GitLab CI YAML template creation
- GitLab Merge Request comment integration
- GitLab API integration for findings display
- GitLab artifact upload

### Excluded
- GitLab self-managed vs SaaS differences (support both)
- GitLab Runner configuration
- GitLab security dashboard integration (future)

---

## 3. Current State

**Status**: GitHub only, no GitLab support

**GitHub Features**: PR comment, Checks, SARIF upload

**Need**: Many organizations use GitLab; need equivalent support.

## 4. Proposed Implementation

### GitLab CI Template

```yaml
# .gitlab/ci/code-to-gate.yml
stages:
  - analysis

code-to-gate-analysis:
  stage: analysis
  image: node:20
  rules:
    - if: $CI_PIPELINE_SOURCE == "merge_request_event"
  script:
    - npm install -g github:RNA4219/code-to-gate
    - code-to-gate scan . --out .qh
    - code-to-gate analyze . --emit all --out .qh
    - code-to-gate readiness . --policy policy.yaml --out .qh
    - code-to-gate export sarif --from .qh --out results.sarif
  artifacts:
    paths:
      - .qh/
    expire_in: 30 days
  after_script:
    - |
      if [ -f .qh/release-readiness.json ]; then
        status=$(jq -r '.status' .qh/release-readiness.json)
        if [ "$status" = "blocked_input" ]; then
          echo "Release blocked by critical findings"
          exit 1
        fi
      fi
```

### GitLab Merge Request Comment

```typescript
// src/gitlab/mr-comment.ts
interface GitLabMrCommentOptions {
  projectId: string;
  mergeRequestIid: number;
  findings: FindingsArtifact;
  readiness?: ReleaseReadinessArtifact;
  gitlabToken: string;
}

async function postMrComment(options: GitLabMrCommentOptions): Promise<void> {
  const comment = generateGitLabComment(options.findings, options.readiness);

  const response = await fetch(
    `https://gitlab.com/api/v4/projects/${options.projectId}/merge_requests/${options.mergeRequestIid}/notes`,
    {
      method: "POST",
      headers: {
        "PRIVATE-TOKEN": options.gitlabToken,
        "Content-Type": "application/json",
      },
      body: JSON.stringify({ body: comment }),
    }
  );

  if (!response.ok) {
    throw new Error(`GitLab API error: ${response.status}`);
  }
}

function generateGitLabComment(
  findings: FindingsArtifact,
  readiness?: ReleaseReadinessArtifact
): string {
  let comment = "## code-to-gate Analysis Results\n\n";

  comment += `| Metric | Count |\n|---|:---:|\n`;
  comment += `| Total Findings | ${findings.findings.length} |\n`;
  comment += `| Critical | ${findings.findings.filter(f => f.severity === "critical").length} |\n`;
  comment += `| High | ${findings.findings.filter(f => f.severity === "high").length} |\n\n`;

  if (readiness) {
    comment += `**Status**: ${readiness.status}\n`;
  }

  comment += `\n---\n*Generated by code-to-gate*`;

  return comment;
}
```
